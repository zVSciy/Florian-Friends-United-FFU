# Albania Receives Ultimatum
country_event = {
	id = italy.3
	title = italy.3.t
	desc = italy.3.d
	picture = GFX_report_event_albanian_king_zog
	
	is_triggered_only = yes
	
	option = { # Yield
		name = italy.3.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					OR = {
						is_major = yes
						tag = YUG
					}
					NOT = { tag = ITA }
					OR = {
						has_guaranteed = ALB
						is_in_faction_with = ALB
					}
				}
			}
			modifier = {
				factor = 0.5
				OR = {
					has_government = communism
					has_government = fascism #glorious Albania will never yield! #Actually, there is nothing glorious to be found in fascism ;)
				}
			}
			modifier = {
				factor = 5
				is_subject_of = FROM
			}
		}
		ITA = {
			country_event = { days = 1 id = italy.1 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Resist
		name = italy.3.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = FROM
						FROM = { is_faction_leader = yes }
					}
					is_historical_focus_on = yes
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = FROM }
								has_guaranteed = ROOT
							} 
						}
					}
				}
			}
			modifier = {
				add = 100
				is_in_faction = yes
				NOT = { is_in_faction_with = FROM }
			}
			modifier = {
				add = 100
				is_subject = yes
				NOT = { is_subject_of = FROM }
			}
		}
		ITA = {
			effect_tooltip = {
				create_wargoal = {
					type = take_state_focus
					target = ALB
					generator = { 44 }
				}
			}
			country_event = { days = 1 id = italy.2 }
		}
	}
}
