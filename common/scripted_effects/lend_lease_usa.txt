# ── INIT ──────────────────────────────────────────────────────
lend_lease_init = {
    set_variable = { lend_lease_target_nation = 0 }
    clr_country_flag = ll_window_open
    clr_country_flag = ll_package_just_sent
}

# ── PACKAGE SENDERS ───────────────────────────────────────────
# Each one branches on lend_lease_target_nation and gives
# equipment directly to the target country's stockpile

ll_send_pkg_infantry = {
    if = {
        limit = { check_variable = { var = lend_lease_target_nation value = 1 compare = equals } }
        add_equipment_to_stockpile = { type = infantry_equipment amount = -1000 producer = ROOT }
        ENG = { add_equipment_to_stockpile = { type = infantry_equipment amount = 1000 producer = FROM } }
