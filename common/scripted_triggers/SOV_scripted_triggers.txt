		limit = {
			has_completed_focus = SOV_secure_new_conquests
		}

	}

		limit = {
			has_completed_focus = SOV_improve_stalin_line
		}
		surrender_progress > 0.22 
	}
	else = {
		surrender_progress > 0.22
	}
	casualties > 500000
